##5. 将item_price转换为浮点数

dollarizer = lambda x: float(x[1:-1])
chipo['item_price'] = chipo['item_price'].apply(dollarizer)

##6. 在该数据集对应的时期内， 收入（revenue） 是多少

chipo['sub_total'] = round(chipo['item_price'] * chipo['quantity'], 2)
